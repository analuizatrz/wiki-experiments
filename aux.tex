Atributos de tamanho

\begin{enumerate}
    \item Quantidade de caracteres, incluindo espaços. %\item Char Count
	\item Quantidade de palavras %\item Word Count
	\item Quantidade de frases %\item Phrase count
\end{enumerate}

Atributos de estrutura

\begin{enumerate}
    \item Quantidade de seções: um bom artigo é organizado em seções, sendo esperado em particular na Wikipédia que o artigo contenha introdução, sumário, lista de referências e ligações externos~\cite{Dalip:2009}.%\item Section count
	\item Tamanho médio das seções: é razão entre o número de caracteres das seções e a quantidade de seções. %\item Mean section size
    % \item [faltou] Mean paragraph size (tsmps)
	\item Tamanho médio das subseções %\item Mean subsection size
	\item Tamanho da menor seção: ajuda a detectar tamanho incomum, como seções vazias ou muito pequenas %\item Shortest section size
	\item Tamanho da maior seção: ajuda a detectar tamanho incomum, como seções que deveriam ser divididas %\item Largest section size
	\item Desvio padrão do tamanho das seções: representa quão balanceadas são as seções%\item Standard deviation of the section size
	\item Quantidade de subseções: um bom artigo é organizado em seções, sendo esperado em particular na Wikipédia que o artigo contenha introdução, sumário, lista de referências e ligações externos~\cite{Dalip:2009}.%\item Subsection count
    %\item [faltou] Mean of subsections per section (tsmsbs)
    %\item [faltou] Abstract size (tsas)
    %\item [faltou] Citation count (tsctc)
    %\item [faltou] Citation count per text length (tsctcl )
    %\item [faltou] Citation count per section (tsctcs)
    %\item [faltou] Links per text length (tsltl )
    %\item [faltou] External link count (tsxlc)
    %\item [faltou] External link per section (tsxls)
	\item Quantidade de imagens: o uso de figura contribui para criar um conteúdo mais claro e visualmente agradável.%\item Images count
	\item Quantidade de imagens por seção %\item Images per section
\end{enumerate}
